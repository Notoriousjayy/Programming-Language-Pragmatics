\centerline{\bf Backus-Naur Form}

\vskip 1cm
$$\vbox{\settabs 2 \columns
	\+\bf Symbol& \bf Meaning\cr
	\+ $|$ & or\cr
	\+ $*$ & zero or more repetitions of the symbol on the left\cr
	\+ $+$ & one or more repetitions of the symbol on the left\cr
	\+\cr}$$
